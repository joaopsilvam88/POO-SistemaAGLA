/*Aluno*/
create table aluno(
    idaluno integer primary key autoincrement,
    matricula varchar(20) not null,
    ano varchar (4) not null    
);

/*Professor*/
create table professor(
    idprofessor integer primary key autoincrement,
    matricula varchar(20) not null,
    lattes varchar (100) not null    
);

/*Dados Pessoais*/
create table dadosPessoais(
    iddadosPessoais integer primary key autoincrement,
    nome VARCHAR(50) not null,
    sexo CHAR(1) not null,
    rg CHAR(13) unique,
    cpf CHAR(11) unique,    
    dataNascimento char(8),
    email VARCHAR(100) unique,/*so existe um email*/
    id_aluno integer,
    foreign key(id_aluno)
    references aluno(idaluno)
);

CREATE TABLE endereco(
    idendereco integer primary key autoincrement,
    estado char(2) not null,
    cidade varchar(50) not null,
    logradouro varchar(30) not null, /*eh obrigatorio*/
    numero integer (4) not null,
    bairro varchar(30) not null,
    complemento varchar (50),
    id_dadis int unique,/*<-de onde vem, e ele eh unique porque nao se repete*/
    foreign key(id_cliente)
    references cliente(idcliente)
    );
